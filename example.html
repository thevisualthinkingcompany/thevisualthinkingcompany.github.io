7 Example

7.1 Satellite

random_lat <- runif(1, min = 48.3, max = 60.0)
random_lng <- runif(1, min = -139.0, max = -114.0)
random_zoom <- runif(1, min = 8, max = 10)
leaflet() %>%
  
 addProviderTiles(providers$Esri.WorldImagery) %>% 
  setView(lng = random_lng, lat = random_lat, zoom = random_zoom) %>%
  addMarkers(lng = random_lng, lat = random_lat, popup = "Random Location in BC")
